# Erste Installation und Setup
npx claude-flow@alpha init --monitoring

# Hive Mind Setup mit Wizard (empfohlen)
npx claude-flow@alpha hive-mind wizard


.yaml erstellen und speichern

# Option A: Direkt mit YAML starten (EMPFOHLEN)
claude-flow swarm "Erstelle PowerPoint-Präsentation PECO Produktkonfigurator 2.0" --config peco_swarm_config.yaml --claude

# Option B: Mit Hive-Mind und YAML
claude-flow hive-mind spawn "PECO Produktkonfigurator PowerPoint erstellen" --config peco_swarm_config.yaml --claude

# Option C: Mit UI und Monitoring
claude-flow start --ui --swarm --config peco_swarm_config.yaml --monitoring
