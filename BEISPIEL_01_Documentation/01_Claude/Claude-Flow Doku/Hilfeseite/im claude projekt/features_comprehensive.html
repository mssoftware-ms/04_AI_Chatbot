What is Model Context Protocol (MCP)?

MCP is an open standard by Anthropic providing a unified interface between AI models and external tools/data sources. Instead of building separate integrations for each tool-model combination, MCP offers a standardized communication layer.

JavaScript API Integration Example ▼ 
// MCP Tool Integration Example
async function orchestrateProject() {
  // 1. Initialize swarm
  await mcp__claude_flow__swarm_init({
    topology: "hierarchical",
    strategy: "auto",
    maxAgents: 10
  });

  // 2. Spawn specialized agents
  const agents = await Promise.all([
    mcp__claude_flow__agent_spawn({
      type: "coordinator",
      name: "ProjectLead"
    }),
    mcp__claude_flow__agent_spawn({
      type: "coder", 
      name: "BackendDev",
      capabilities: ["python", "fastapi"]
    }),
    mcp__claude_flow__agent_spawn({
      type: "tester",
      name: "QAEngineer"
    })
  ]);

  // 3. Orchestrate tasks
  await mcp__claude_flow__task_orchestrate({
    task: "Build REST API with authentication",
    strategy: "parallel",
    priority: "high",
    agents: agents.map(a => a.id)
  });

  // 4. Store progress in memory
  await mcp__claude_flow__memory_store({
    key: "project-status",
    value: { phase: "development", progress: 0.25 },
    namespace: "api-project"
  });

  // 5. Monitor performance
  const metrics = await mcp__claude_flow__performance_monitor({
    metrics: ["task-completion", "agent-efficiency"],
    interval: 5000
  });

  return metrics;
}

Practical Hook Example: Automatic Code Formatting ▼ 
// hooks/pre-tool.js - Automatic Code Formatting
module.exports = async function preToolHook(context) {
  const { tool, args, agent } = context;
  
  // Auto-format on file writes
  if (tool === 'Write' || tool === 'Edit') {
    const filePath = args.path;
    const content = args.content;
    
    // Language-specific formatting
    if (filePath.endsWith('.py')) {
      args.content = await formatPython(content);
      log(`Formatted Python file: ${filePath}`);
    } else if (filePath.endsWith('.js') || filePath.endsWith('.ts')) {
      args.content = await formatJavaScript(content);
      log(`Formatted JavaScript file: ${filePath}`);
    }
    
    // Linting validation
    const lintErrors = await lint(filePath, args.content);
    if (lintErrors.length > 0) {
      return {
        allow: false,
        reason: `Linting errors: ${lintErrors.join(', ')}`
      };
    }
  }
  
  return { allow: true };
}
